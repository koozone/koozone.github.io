메소드 사용법

// 대상 element의 변수 값 가져오기 kz.element.elValue({name}); // 대상 element의 변수 값 적용하기 kz.element.elValue({name}, {value});
  • {name} : 대상 element의 input name
  • {value} : 적용 변수 값

이벤트 사용법

/* jquery */ // focus 이벤트 $('[name={name}]').on('focus', (event) => { console.log(event.target.name, 'focus'); }); // click 이벤트 $('[name={name}]').on('click', (event) => { console.log(event.target.name, 'click'); }); // change 이벤트 $('[name={name}]').on('changeEl', (event) => { console.log(event.target.name, 'change : ', kz.element.elValue(event.target.name)); }); /* javascript */ // change 이벤트 document.querySelector('[name={name}]').addEventListener('changeEl', (event) => { });
  • {name} : 대상 element의 input name
